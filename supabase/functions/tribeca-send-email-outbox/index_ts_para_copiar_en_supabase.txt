// Tribeca Aula · versión 64
// Edge Function para enviar emails pendientes de public.email_outbox mediante Resend.
// Cambio v64: si la web invoca la función con event_type = "message" o "calendar",
// solo se envían emails pendientes de ese tipo. Así un mensaje no arrastra avisos
// antiguos de calendario que hubieran quedado pendientes en la cola.

import { serve } from "https://deno.land/std@0.224.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2.45.4";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
  "Access-Control-Allow-Methods": "POST, OPTIONS",
};

serve(async (req) => {
  if (req.method === "OPTIONS") return new Response("ok", { headers: corsHeaders });
  if (req.method !== "POST") {
    return json({ ok: false, error: "Method not allowed" }, 405);
  }

  const supabaseUrl = Deno.env.get("SUPABASE_URL");
  const serviceRole = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY");
  const resendKey = Deno.env.get("RESEND_API_KEY");
  const from = Deno.env.get("TRIBECA_EMAIL_FROM") || "Tribeca Aula <onboarding@resend.dev>";

  if (!supabaseUrl || !serviceRole || !resendKey) {
    return json({
      ok: false,
      error: "Missing SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY or RESEND_API_KEY",
    }, 500);
  }

  let requestedType = "";
  try {
    const body = await req.json().catch(() => ({}));
    requestedType = String(body?.event_type || body?.type || body?.reason || "").trim().toLowerCase();
  } catch (_) {
    requestedType = "";
  }

  const allowedTypes = new Set(["message", "calendar", "announcement", "material", "generic"]);
  const eventType = allowedTypes.has(requestedType) ? requestedType : "";

  const supabase = createClient(supabaseUrl, serviceRole, { auth: { persistSession: false } });

  let query = supabase
    .from("email_outbox")
    .select("id, recipient_email, subject, body, event_type")
    .eq("status", "pending")
    .order("created_at", { ascending: true })
    .limit(25);

  if (eventType) {
    query = query.eq("event_type", eventType);
  }

  const { data: rows, error } = await query;

  if (error) return json({ ok: false, error: error.message }, 500);

  let sent = 0;
  let failed = 0;

  for (const row of rows || []) {
    try {
      const response = await fetch("https://api.resend.com/emails", {
        method: "POST",
        headers: {
          "Authorization": `Bearer ${resendKey}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          from,
          to: [row.recipient_email],
          subject: row.subject,
          text: row.body,
          html: `<pre style="font-family: Arial, sans-serif; white-space: pre-wrap; line-height: 1.5; color: #172018;">${escapeHtml(row.body)}</pre>`,
        }),
      });

      if (!response.ok) {
        const errText = await response.text();
        throw new Error(errText || `Resend error ${response.status}`);
      }

      await supabase
        .from("email_outbox")
        .update({ status: "sent", sent_at: new Date().toISOString(), error: null })
        .eq("id", row.id);

      sent++;
    } catch (e) {
      failed++;
      await supabase
        .from("email_outbox")
        .update({ status: "error", error: String(e?.message || e) })
        .eq("id", row.id);
    }
  }

  return json({
    ok: true,
    event_type: eventType || "all",
    sent,
    failed,
    checked: rows?.length || 0,
  });
});

function json(payload: unknown, status = 200) {
  return new Response(JSON.stringify(payload), {
    status,
    headers: { ...corsHeaders, "Content-Type": "application/json" },
  });
}

function escapeHtml(value: string) {
  return String(value || "")
    .replaceAll("&", "&amp;")
    .replaceAll("<", "&lt;")
    .replaceAll(">", "&gt;")
    .replaceAll('"', "&quot;")
    .replaceAll("'", "&#039;");
}
